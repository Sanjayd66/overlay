// View Panel Script const fbURL = "https://bgmi-overlay-default-rtdb.firebaseio.com/liveData.json"; async function updateView() { try { const response = await fetch(fbURL); const data = await response.json(); if (data) { // 1. Update Name document.getElementById('teamName').innerText = data.name; // 2. Update Logo (Aapke logo folder ka path sahi hona chahiye) // Agar aapke folder mein 1.png, 2.png hai toh: document.getElementById('teamLogo').src = `logos/${data.logo}.png`; // 3. Update Background Color document.getElementById('logoBG').style.backgroundColor = data.sbg; } } catch (e) { console.error("Data fetch error:", e); } } // Har 1 second mein check karega setInterval(updateView, 1000);
